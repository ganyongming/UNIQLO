　　//冒泡排序
public static void main(String[] args) {
		 int[] arr = new int[]{5,8,4,2,7,9,5};
		 for(int i=0;i<arr.length;i++){
			 for(int j=i;j<arr.length-1;j++){
			   if(arr[i]>arr[j+1]){
				   int t = arr[i];
				   arr[i] = arr[j+1];
				   arr[j+1] = t;
			   }
			 }
		}
		 for(int i:arr){
			 System.out.print(i+" ");
		 }
					
	}
